# 枚举

# 枚举变量的定义

1.先定义枚举类型，然后定义枚举变量


enum DAY
{
    MON=1, TUE, WED, THU, FRI, SAT, SUN
};

enum DAY day;

2.定义枚举类型的同时定义枚举变量


enum DAY
{
    MON=1, TUE, WED, THU, FRI, SAT, SUN
} day;

3.省略枚举名称，直接定义枚举变量


enum
{
    MON=1, TUE, WED, THU, FRI, SAT, SUN
} day;

# 指针

# 指针定义

指针是一个变量，值为另一个变量的地址，即内存位置的直接地址。

&：获得变量的地址 *：用来访问指针的值所表示的地址上的变量 变量的值是内存的地址 - 普通变量的值是实际的值 - 指针变量的值是具有实际值的变量的地址

int a[] = {1, 2}; int b[] --> int* const b; //静态变量是 const 的指针，所以不能被赋值 int *p = a; //*运算符可以对指针做，也可以对数组做

char* s = "Hello"; //s 是一个指针，初始化为指向一个字符串常量，实际上是 const char* s 如果需要修改字符串，应该使用数组 char s[] = "Hello";

数组：这个字符串在这里 - 作为本地变量空间自动被回收 指针：这个字符串不知道在哪里 - 处理参数 - 动态分配空间

char* t = "title"; char* s = t; //s和t指向同一

# 普通指针


int main() {
    int a = 0;
    int* b;

    b = &a;

    printf("%p\n", b);
    printf("%d\n", a);

    *b = 10;

    printf("%d\n", *b);
    printf("%d\n", a);
}

# 函数指针


int max(int a, int b) {
    return a > b ? a : b;
}

int main() {
    int (*p)(int, int) = &max; // &可以省略
    int a = 0;
    int b = 3;

    printf("%d", p(a, b));
}

# 回调函数


int max(int a, int b) {
    return a > b ? a : b;
}

void printA(int a) {
    printf("%d", a);
}

void test(int a, int b, void (*t)(int)) {
    int c = max(a, b);

    t(c);
}

int main() {
    test(0, 3, printA);
}

# 结构


typedef struct
{
    int a;
    double b;
    char c;
} Test;

//包含指向 Test 结构的指针
typedef struct {
    char str[100];
    Test* test;
} A;

//包含指向 Node 结构的指针
struct NODE
{
    char string[100];
    struct NODE* next_node;
};




struct Test
{
    int a;
    double b;
    char c;
};


void printTest(struct Test t[]);
void printT(struct Test *t);

int main() {
    struct Test t = {1, 2.0, 'c'};

    printT(&t);
}

void printTest(struct Test t[]) {
    printf("a:%d\n", t[0].a);
    printf("b:%f\n", t[0].b);
    printf("c:%d\n", t[0].c);
}

void printT(struct Test *t) {
    printf("a:%d\n", t -> a);
    printf("b:%f\n", t -> b);
    printf("c:%d\n", t -> c);
}

# 位域定义

一个字节中的二进位划分为几个不同的区域，并说明每个区域的位数。每个域有一个域名，允许在程序中按域名进行操作。这样就可以把几个不同的对象用一个字节的二进制位域来表示
