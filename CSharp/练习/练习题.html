# 闰年

# 思路

闰年的判定（满足其一即可）：

  • 年份能被 400 整除
  • 年份能被 4 整除但不能被 100 整除

# 代码

public static bool isLeapYear(int year) {
    return (year % 4 == 0 && year % 100 != 0) || year % 400 == 0;
}

# 水仙花

# 水仙花定义

  • 三位数
  • 每个位上的数字的 3次幂之和等于它本身

# 水仙花代码

public static bool IsNarcissisticNumber(int num) {
    if (num < 100 || num > 999) {
        return false;
    }

    int hundredNum = num / 100;
    int tenNum = (num % 100) / 10;
    int pieceNum = num % 10;

    if (num == Math.Pow(hundredNum, 3) + Math.Pow(tenNum, 3) + Math.Pow(pieceNum, 3)) {
            return true;
    }

    return false;
}
